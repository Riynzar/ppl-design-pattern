#include <iostream>
#include <string>
#include <thread>
#include <mutex>
#include <chrono>

// ── JADWAL ───────────────────────────────────────────────────
struct Jadwal {
    std::string idJadwal;
    std::string waktu;
    bool        terisi;
};

// ── BOOKING SERVICE dengan Mutex (Lock) ──────────────────────
class BookingService {
private:
    std::mutex  mtx;          // Lock untuk mutual exclusion
    Jadwal      jadwal;

public:
    BookingService(std::string id, std::string waktu) {
        jadwal = { id, waktu, false };
    }

    void pesanJadwal(std::string mahasiswa) {
        std::cout << "[" << mahasiswa << "] Mencoba memesan slot "
                  << jadwal.idJadwal << " (" << jadwal.waktu << ")..."
                  << std::endl;

        // Kunci akses — hanya 1 thread yang boleh masuk
        std::lock_guard<std::mutex> lock(mtx);

        std::cout << "[" << mahasiswa << "] Sistem DIKUNCI. Mengecek ketersediaan..."
                  << std::endl;

        // Simulasi proses pengecekan database
        std::this_thread::sleep_for(std::chrono::milliseconds(500));

        if (!jadwal.terisi) {
            jadwal.terisi = true;
            std::cout << "[" << mahasiswa << "] BERHASIL! Slot berhasil dipesan." << std::endl;
        } else {
            std::cout << "[" << mahasiswa << "] GAGAL! Slot sudah dipesan mahasiswa lain." << std::endl;
        }

        std::cout << "[" << mahasiswa << "] Kunci sistem DILEPAS." << std::endl << std::endl;
        // Lock otomatis dilepas saat lock_guard keluar scope
    }
};

int main() {
    std::cout << std::endl;
    std::cout << "  CONCURRENCY PATTERN — TutorKu Booking System" << std::endl;
    std::cout << "  Simulasi 2 mahasiswa memesan slot yang sama" << std::endl;
    std::cout << "  secara bersamaan" << std::endl;
    std::cout << std::endl;
    std::cout << "======================================" << std::endl;
    std::cout << "  Slot   : J001 — Rabu, 14.00-15.00" << std::endl;
    std::cout << "  Tutor  : Pak Budi (Struktur Data)" << std::endl;
    std::cout << "======================================" << std::endl << std::endl;

    BookingService booking("J001", "Rabu 14.00-15.00");

    // 2 mahasiswa mencoba memesan slot yang sama bersamaan
    std::thread t1(&BookingService::pesanJadwal, &booking, "Angela Vania");
    std::thread t2(&BookingService::pesanJadwal, &booking, "Berwyn Rafif");

    t1.join();
    t2.join();

    std::cout << "======================================" << std::endl;
    std::cout << "  Hasil akhir: Slot J001 = "
              << "TERISI" << std::endl;
    std::cout << "  Race condition berhasil dicegah!" << std::endl;
    std::cout << "======================================" << std::endl << std::endl;

    return 0;
}
